1336  int
1337  MC_Parser::parse ()
1338  {
1339  // State.
1340  int yyn;
1342  int yylen = 0;
1343 
1344  // Error handling.
1345  int yynerrs_ = 0;
1346  int yyerrstatus_ = 0;
1347 
1349  symbol_type yyla;
1350 
1352  stack_symbol_type yyerror_range[3];
1353 
1355  int yyresult;
1356 
1357  // FIXME: This shoud be completely indented. It is not yet to
1358  // avoid gratuitous conflicts when merging into the master branch.
1359  try
1360  {
1361  YYCDEBUG << "Starting parse" << std::endl;
1362 
1363 
1364  /* Initialize the stack. The initial state will be set in
1365  yynewstate, since the latter expects the semantical and the
1366  location values to have been already stored, initialize these
1367  stacks with a primary value. */
1368  yystack_.clear ();
1369  yypush_ (YY_NULLPTR, 0, yyla);
1370 
1371  // A new symbol was pushed on the stack.
1372  yynewstate:
1373  YYCDEBUG << "Entering state " << yystack_[0].state << std::endl;
1374 
1375  // Accept?
1376  if (yystack_[0].state == yyfinal_)
1377  goto yyacceptlab;
1378 
1379  goto yybackup;
1380 
1381  // Backup.
1382  yybackup:
1383 
1384  // Try to take a decision without lookahead.
1385  yyn = yypact_[yystack_[0].state];
1386  if (yy_pact_value_is_default_ (yyn))
1387  goto yydefault;
1388 
1389  // Read a lookahead token.
1390  if (yyla.empty ())
1391  {
1392  YYCDEBUG << "Reading a token: ";
1393  try
1394  {
1395  yyla.type = yytranslate_ (yylex (&yyla.value, &yyla.location));
1396  }
1397  catch (const syntax_error& yyexc)
1398  {
1399  error (yyexc);
1400  goto yyerrlab1;
1401  }
1402  }
1403  YY_SYMBOL_PRINT ("Next token is", yyla);
1404 
1405  /* If the proper action on seeing token YYLA.TYPE is to reduce or
1406  to detect an error, take that action. */
1407  yyn += yyla.type_get ();
1408  if (yyn < 0 || yylast_ < yyn || yycheck_[yyn] != yyla.type_get ())
1409  goto yydefault;
1410 
1411  // Reduce or error.
1412  yyn = yytable_[yyn];
1413  if (yyn <= 0)
1414  {
1415  if (yy_table_value_is_error_ (yyn))
1416  goto yyerrlab;
1417  yyn = -yyn;
1418  goto yyreduce;
1419  }
1420

1422  if (yyerrstatus_)
1423  --yyerrstatus_;

1426  yypush_ ("Shifting", yyn, yyla);
